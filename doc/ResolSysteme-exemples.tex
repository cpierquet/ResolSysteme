\section{Déterminant d'une matrice, 2x2 ou 3x3 ou 4x4}

\begin{ShowCodeTeX}

$\det(A)=\DetMatrice(1,2 § 3,4)$.
\end{ShowCodeTeX}

\begin{ShowCodeTeX}

$\det(A)=\DetMatrice[dec](-1,0.5 § -1/2,4)$.
\end{ShowCodeTeX}

\begin{ShowCodeTeX}
